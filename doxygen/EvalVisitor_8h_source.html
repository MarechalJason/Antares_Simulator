52  threshold_(threshold)

56  double operator()(double lhs, double rhs) const
57  {

214 template<typename BinaryOp>
215 std::vector<double> computeBinaryOperation(const std::vector<double>& lhs,
216  const std::vector<double>& rhs,
217  BinaryOp op)
218 {
219  if (lhs.size() != rhs.size())
220  {
221  std::ostringstream errorMsg;
222  errorMsg << "Failed to compute binary operation: vectors have different sizes ("
223  << lhs.size() << " and " << rhs.size() << ").";
224  throw VectorsMismatchSize(errorMsg.str());
225  }
226 
227  std::vector<double> result(lhs.size());
228  for (size_t i = 0; i < lhs.size(); ++i)
229  {
230  result[i] = op(lhs[i], rhs[i]);
231  }
232  return result;
233 }

272 template<class Operation>
273 EvaluationResult applyOperation(const std::vector<EvaluationResult>& in, Operation op)
274 {
275  if (in.size() < 2)
276  {
277  throw std::invalid_argument("Expected at least two EvaluationResult");
278  }
279  const size_t size = getMaxSize(in);
280  std::vector<double> values(size);
281  std::vector<double> row(in.size());
282 
283  for (size_t timeIndex = 0; timeIndex < size; ++timeIndex)
284  {
285  for (size_t evalIndex = 0; evalIndex < in.size(); ++evalIndex)
286  {
287  const auto& evalResult = in[evalIndex];
288  row[evalIndex] = evalResult.value(timeIndex);
289  }
290  values[timeIndex] = op(row);
291  }
292  if (size > 1)
293  {
294  return EvaluationResult(values);
295  }
296 
297  return EvaluationResult(values.at(0));
298 }
