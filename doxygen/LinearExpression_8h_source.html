98 void add_maps(MapType& left, const MapType& right, UnaryOp op = std::identity{})
99 {
100  for (const auto& [key, value]: right)
101  {
102  auto it = left.find(key);
103  if (it != left.end())
104  {
105  // Key exists in left, add the values
106  it->second += op(value);
107  }
108  else
109  {
110  // Key does not exist in left, insert the pair
111  left.emplace(key, op(value));
112  }
113  }
114 }
