300 template<class R, class... Args>
301 std::vector<R> NodeVisitor<R, Args...>::visitChildrenNodes(const Nodes::ParentNode* node)
302 {
303  std::vector<R> result;
304  std::ranges::transform(node->getOperands(),
305  std::back_inserter(result),
