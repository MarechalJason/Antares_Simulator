57 template<class StringT, class D>
58 bool ApplyToMatrix(uint& errors,
59  StringT& logprefix,
60  D& data,
61  const TSNumberData::MatrixType::ColumnType& years,
62  uint tsGenMax)
63 {
64  bool ret = true;
65 
66  // In this case, m.height represents the total number of years
67  const uint nbYears = data.timeseriesNumbers.height();
68  // The matrix m has only one column
69  auto& target = data.timeseriesNumbers;
70 
71  for (uint y = 0; y != nbYears; ++y)
72  {
73  if (years[y] != 0)
74  {
75  // The new TS number
76  uint32_t tsNum = years[y] - 1;
77 
78  // When the TS-Generators are not used
79  if (!CheckValidity(tsNum, data, tsGenMax))
80  {
81  if (errors <= maxErrors)
82  {
83  if (++errors == maxErrors)
84  {
85  logs.warning() << "scenario-builder: ... (skipped)";
86  }
87  else
88  {
89  logs.warning() << "scenario-builder: " << logprefix
90  << "value out of bounds for the year " << (y + 1);
91  }
92  }
93  ret = false;
94  continue;
95  }
96  // Ok, assign. The value provided by the interface is user-friendly
97  // and starts from 1.
98  target[y] = tsNum;
99  }
100  }
101 
102  return ret;
103 }
