112  template<class VCardT>
113  void buildDigest(SurveyResults& report, int digestLevel, int dataLevel) const
114  {
115  const bool isCluster = (VCardT::categoryFileLevel & Category::FileLevel::de)
116  || (VCardT::categoryFileLevel & Category::FileLevel::de_res);
117  const bool isBindingConstraint = VCardT::categoryFileLevel & Category::FileLevel::bc;
118  const bool isDigest = digestLevel & Category::digestAllYears;
119  if ((dataLevel & Category::DataLevel::area || dataLevel & Category::DataLevel::setOfAreas)
120  && isDigest && !isCluster && !isBindingConstraint)
121  {
122  assert(report.data.columnIndex < report.maxVariables && "Column index out of bounds");
123 
124  report.captions[0][report.data.columnIndex] = report.variableCaption;
125  report.captions[1][report.data.columnIndex] = report.variableUnit;
126  report.captions[2][report.data.columnIndex] = (report.variableCaption == "LOLP")
127  ? "values"
128  : "EXP";
129 
130  // Precision
131  report.precision[report.data.columnIndex] = PrecisionToPrintfFormat<
132  VCardT::decimal>::Value();
133  // Value
134  report.values[report.data.columnIndex][report.data.rowIndex] = avgdata.year;
135  // Non applicability
136  report.digestNonApplicableStatus[report.data.rowIndex][report.data.columnIndex]
137  = *report.isCurrentVarNA;

150  void InternalExportValues(SurveyResults& report, const std::vector<HighPrecision>& array) const
151  {
152  assert(report.data.columnIndex < report.maxVariables && "Column index out of bounds");
153 
154  // Caption
155  report.captions[0][report.data.columnIndex] = report.variableCaption;
156  report.captions[1][report.data.columnIndex] = report.variableUnit;
157  report.captions[2][report.data.columnIndex] = (report.variableCaption == "LOLP") ? "values"
158  : "EXP";
159  // Precision
160  report.precision[report.data.columnIndex] = PrecisionToPrintfFormat<
161  VCardT::decimal>::Value();
162  // Non applicability
163  report.nonApplicableStatus[report.data.columnIndex] = *report.isCurrentVarNA;
164 
165  // Values
166  switch (PrecisionT)
167  {
168  case Category::hourly:
169  {
170  for (uint h = 0; h < HOURS_PER_YEAR; ++h)
171  {
172  report.values[report.data.columnIndex][h] = array[h];
173  }
174  break;
175  }
176  case Category::daily:
177  {
178  for (uint d = 0; d < DAYS_PER_YEAR; ++d)
179  {
180  report.values[report.data.columnIndex][d] = array[d];
181  }
182  break;
183  }
184 
185  case Category::weekly:
186  {
187  for (uint w = 0; w < WEEKS_PER_YEAR; ++w)
188  {
189  report.values[report.data.columnIndex][w] = array[w];
190  }
191  break;
192  }
193 
194  case Category::monthly:
195  {
196  for (uint m = 0; m < MONTHS_PER_YEAR; ++m)
197  {
198  report.values[report.data.columnIndex][m] = array[m];
199  }
200  break;
201  }
202  case Category::annual:
203  {
204  *(report.values[report.data.columnIndex]) = array[0];
205  break;
206  }
207 
208  default:
209  logs.warning() << "Category not found for variable: " << report.variableCaption;
210  break;
211  }
212 
213  // Next column index
214  ++report.data.columnIndex;
215  }
