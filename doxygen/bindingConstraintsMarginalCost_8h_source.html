63 /*
64  Marginal cost associated to binding constraints :
65  Suppose that the BC is hourly,
66  - if binding constraint is not saturated (rhs is not reached) for a given hour, the value is 0;
67  - if binding constraint is saturated (rhs is reached), the value is the total benefit (�/MW) for
68  the system that would result in increasing the BC's rhs of 1 MW.
69 */

207  void weekBegin(State& state)
208  {
209  if (isInitialized())
210  {
211  auto numSpace = state.numSpace;
212  // For daily binding constraints, getting daily marginal price
213  using namespace Data;
214  switch (associatedBC_->type())
215  {
216  case BindingConstraint::typeHourly:
217  case BindingConstraint::typeUnknown:
218  case BindingConstraint::typeMax:
219  break;
220 
221  case BindingConstraint::typeDaily:
222  {
223  int dayInTheYear = state.weekInTheYear * 7;
224  for (int dayInTheWeek = 0; dayInTheWeek < 7; dayInTheWeek++)
225  {
226  pValuesForTheCurrentYear[numSpace].day[dayInTheYear]
227  -= state.problemeHebdo
228  ->ResultatsContraintesCouplantes[associatedBC_][dayInTheWeek];
229 
230  dayInTheYear++;
231  }
232  break;
233  }
234 
235  // For weekly binding constraints, getting weekly marginal price
236  case BindingConstraint::typeWeekly:
237  {
238  uint weekInTheYear = state.weekInTheYear;
239  double weeklyValue = -state.problemeHebdo
240  ->ResultatsContraintesCouplantes[associatedBC_][0];
241 
242  pValuesForTheCurrentYear[numSpace].week[weekInTheYear] = weeklyValue;
243 
244  int dayInTheYear = state.weekInTheYear * 7;
245  for (int dayInTheWeek = 0; dayInTheWeek < 7; dayInTheWeek++)
246  {
247  pValuesForTheCurrentYear[numSpace].day[dayInTheYear] = weeklyValue;
248  dayInTheYear++;
249  }
250  break;
251  }
252  }
253  }
254  NextType::weekBegin(state);
255  }

418  bool isCurrentOutputNonApplicable(int precision) const
419  {
420  using namespace Antares::Data;
421  // The current marginal prices to print becomes non applicable if they have a precision
422  // (hour, day, week, ...) smaller than the associated binding constraint granularity.
423  // Ex : if the BC is daily and we try to print hourly associated marginal prices,
424  // then these prices are set to N/A
425  switch (associatedBC_->type())
426  {
427  case BindingConstraint::typeUnknown:
428  case BindingConstraint::typeMax:
429  return true;
430  default:
431  const auto precision_bc = 1 << (associatedBC_->type() - 1);
432  return precision < precision_bc;
433  }
434  }
