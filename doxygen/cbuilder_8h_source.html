118 class Cycle
119 {
120 public:
121  Cycle(const std::vector<linkInfo*>& linkList, double infinite = 1000000):
122  time(0),
123  loop(linkList),
124  opType(Data::BindingConstraint::opEquality),
125  pInfinite(infinite)
126  {
127  uint columnImpedance = (uint)Antares::Data::fhlImpedances;
128 
129  std::vector<double> impedances;
130  Data::AreaLink* previousLine = linkList[0]->ptr;
131  double currentLineSign = 1;
132  for (auto line = linkList.begin(); line != linkList.end(); line++)
133  {
134  if ((*line)->nImpedanceChanges > 0
135  || ((*line)->type == Antares::Data::atAC && (!(*line)->hasPShiftsEqual)))
136  {
137  opType = Data::BindingConstraint::opBoth;
138  }
139  impedances.push_back((*line)->ptr->parameters[columnImpedance][0]);
140 
141  time = HOURS_PER_YEAR; /*BC loading always expects 8786 values heigth will
142  have to be resized*/
143 
144  if (line == linkList.begin())
145  {
146  currentLineSign = 1;
147  }
148  else
149  {
150  if (previousLine->with == (*line)->ptr->from
151  || previousLine->from
152  == (*line)->ptr->with) //[(A/B),(B/C),... ] or [(A,B),(C/A),... ]
153  {
154  }
155  else if (previousLine->from == (*line)->ptr->from
156  || previousLine->with == (*line)->ptr->with)
157  {
158  currentLineSign *= -1;
159  }
160  else
161  {
162  assert(0 and "links of the loops do not connect or are not in the right order");
163  }
164  }
165 
166  sign.push_back(currentLineSign);
167  previousLine = (*line)->ptr;
168  }
169  State state(impedances, time, pInfinite);
170  states.push_back(state);
171  }
172 
173  State& getState(std::vector<double>& impedances)
174  {
175  std::vector<State>::iterator stIT = std::find_if(states.begin(),
176  states.end(),
177  [&impedances](State& s) -> bool
178  { return s.impedances == impedances; });
179 
180  if (stIT == states.end())
181  {
182  State state(impedances, time, pInfinite);
183  states.push_back(state);
184  stIT = std::find_if(states.begin(),
185  states.end(),
186  [&impedances](State& s) -> bool
187  { return s.impedances == impedances; });
188  }
189  return *stIT;
190  }
191

193  std::vector<double> sign;
194  const std::vector<linkInfo*>& loop;
195  std::vector<State> states;
196  Data::BindingConstraint::Operator opType;
197  double pInfinite;
198 };
