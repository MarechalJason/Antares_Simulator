28 class yearJob
29 {
30 public:
31  yearJob(ISimulation<Impl>* simulation,
32  unsigned int pY,
33  std::map<uint, bool>& pYearsFailed,
34  NumSpaceManager& numspaceManager,
35  randomNumbers& pRandomForParallelYears,
36  Data::Study& pStudy,
37  std::vector<Variable::State>& pStates,
38  bool pYearByYear,
39  Benchmarking::DurationCollector& durationCollector,
40  IResultWriter& resultWriter,
41  ISimulationObserver& simulationObserver,
42  std::mutex& aggregationMutex):
43  simulation_(simulation),
44  y(pY),
45  yearsFailed(pYearsFailed),
46  numspaceManager(numspaceManager),
47  randomForParallelYears(pRandomForParallelYears),
48  study(pStudy),
49  states(pStates),
50  yearByYear(pYearByYear),
51  pDurationCollector(durationCollector),
52  pResultWriter(resultWriter),
53  simulationObserver_(simulationObserver),
54  hydroManagement(study.areas, study.parameters, study.calendar, resultWriter),
55  aggregationMutex(aggregationMutex)
56  {
57  }
58 
59  yearJob(const yearJob&) = delete;
60  yearJob& operator=(const yearJob&) = delete;
61  ~yearJob() = default;
62 
63 private:
64  ISimulation<Impl>* simulation_;
65  unsigned int y;
66  std::map<uint, bool>& yearsFailed;
67  NumSpaceManager& numspaceManager;
68  randomNumbers& randomForParallelYears;
69  Data::Study& study;
70  std::vector<Variable::State>& states;
71  bool yearByYear;
72  Benchmarking::DurationCollector& pDurationCollector;
73  IResultWriter& pResultWriter;
74  ISimulationObserver& simulationObserver_;
75  HydroManagement hydroManagement;
76  std::mutex& aggregationMutex;
77 
78 private:
79  /*
80  ** \brief Log failed week
81  **
82  ** \param y MC Year
83  ** \param study The Antares study
84  ** \param failedWeek List of failing week
85  */
86  void logFailedWeek(int y, const Data::Study& study, const std::list<uint>& failedWeekList)
87  {
88  if (!failedWeekList.empty())
89  {
90  std::stringstream failedWeekStr;
91  std::ranges::copy(failedWeekList, std::ostream_iterator<int>(failedWeekStr, " "));
92 
93  std::string s = failedWeekStr.str();
94  s = s.substr(0, s.length() - 1); // get rid of the trailing space
95 
96  std::string failedStr = failedWeekList.size() != 1 ? " failed at weeks "
97  : " failed at week ";
98 
99  logs.info(); // empty line
100 
101  if (Data::stopSimulation(study.parameters.include.unfeasibleProblemBehavior))
102  {
103  logs.fatal() << "Year " << y + 1 << failedStr << s << ".";
104  }
105  else
106  {
107  logs.warning() << "Year " << y + 1 << failedStr << s << ".";
108  }
109  }
110  }
111 
112 public:
113  void operator()()
114  {
115  Progression::Task progression(study, y, Solver::Progression::sectYear);
116 
117  // Index of the current year in the list of structures
118  uint indexYear = randomForParallelYears.yearNumberToIndex[y];
119 
120  // Getting random tables for this year
121  yearRandomNumbers& randomForCurrentYear = randomForParallelYears.pYears[indexYear];
122 
123  // 1 - Applying random levels for current year
124  auto randomReservoirLevel = randomForCurrentYear.pReservoirLevels;
125 
126  // 2 - Getting the numpspace and scratchMap associated to the current year
127  unsigned numSpace = numspaceManager.getAvailableNumSpace();
128  Yuni::Logs::threadNumber() = numSpace;
129  logs.info() << "Year " << y + 1 << " started";
130 
131  Antares::Data::Area::ScratchMap scratchmap = study.areas.buildScratchMap(numSpace);
132 
133  // 3 - Preparing data related to Clusters in 'must-run' mode
134  prepareClustersInMustRunMode(study, scratchmap, y, Impl::mode);
135 
136  // 4 - Hydraulic ventilation
137  pDurationCollector("hydro_ventilation") << [this, &scratchmap, &randomReservoirLevel]
138  { hydroManagement.makeVentilation(randomReservoirLevel, y, scratchmap); };
139 
140  // Updating the state
141  auto& state = states[numSpace];
142  state.year = y;
143 
144  // 5 - Resetting all variables for the output
145  simulation_->variables.yearBegin(y, numSpace);
146 
147  // 6 - The Solver itself
148  std::list<uint> failedWeekList;
149 
150  OptimizationStatisticsWriter optWriter(pResultWriter, y);
151  bool yearFailed = !simulation_->year(progression,
152  state,
153  numSpace,
154  randomForCurrentYear,
155  failedWeekList,
156  hydroManagement.ventilationResults(),
157  optWriter,
158  pDurationCollector,
159  scratchmap);
160  if (!study.parameters.noOutput)
161  {
162  auto& simTable = simulation_->getSimulationTable(numSpace);
163 
164  auto buffers = simTable.moveBuffers();
165 
166  simulation_->storeYearBuffers(y, std::move(buffers.first), std::move(buffers.second));
167  }
168 
169  // Log failing weeks
170  logFailedWeek(y, study, failedWeekList);
171 
172  simulation_->variables.yearEndBuild(state, y, numSpace);
173 
174  // 7 - End of the year, this is the last stade where the variables can retrieve
175  // their data for this year.
176  simulation_->variables.yearEnd(y, numSpace);
177 
178  // 8 - Spatial clusters
179  // Notifying all variables to perform spatial aggregates.
180  // This must be done only when all variables have finished to compute their
181  // data for the year.
182  simulation_->variables.yearEndSpatialAggregates(simulation_->variables, y, numSpace);
183 
184  // 9 - Write results for the current year
185  if (yearByYear)
186  {
187  pDurationCollector("yby_export") << [this, &numSpace]
188  {
189  // Before writing, some variable may require minor modifications
190  simulation_->variables.beforeYearByYearExport(y, numSpace);
191  // writing the results for the current year into the output
192  simulation_->writeResults(false, y, numSpace); // false for synthesis
193  };
194  }
195 
196  // 10 - Synthesis results
197  // Computing the summary : adding the contribution of MC years
198  // previously computed in parallel
199  aggregationMutex.lock();
200 
201  yearsFailed[y] = yearFailed;
202 
203  pDurationCollector("synthesis_compute") << [this, &numSpace, &state]
204  {
205  simulation_->variables.computeSummary(y, numSpace);
206 
207  // Computing summary of spatial aggregations
208  simulation_->variables.computeSpatialAggregatesSummary(simulation_->variables,
209  y,
210  numSpace);
211 
212  // Computes statistics on annual (system and solution) costs, to be printed in output
213  // into separate files
214  simulation_->computeAnnualCostsStatistics(state);
215  };
216 
217  logs.debug() << "year " << y + 1 << " ended and returned numSpace " << numSpace;
218  numspaceManager.freeNumSpace(numSpace);
219  simulation_->incrementProgression(progression);
220 
221  aggregationMutex.unlock();
222 
223  } // End of onExecute() method
224 };

270 template<class ImplementationType>
271 void ISimulation<ImplementationType>::run()
272 {
273  pNbMaxPerformedYearsInParallel = study.maxNbYearsInParallel;
274 
275  // Initialize all data
276  ImplementationType::variables.initializeFromStudy(study);
277 
278  study.parameters.variablesPrintInfo.computeMaxColumnsCountInReports(study.setsOfAreas);
279 
280  logs.info() << "Allocating resources...";
281 
282  // Memory usage
283  {
284  Variable::PrintInfosStdCout c;
285  ImplementationType::variables.template provideInformations<Variable::PrintInfosStdCout>(c);
286  }
287 
288  ImplementationType::setNbPerformedYearsInParallel(pNbMaxPerformedYearsInParallel);
289 
290  if (settings.tsGeneratorsOnly)
291  {
292  // Only the preprocessors can be used
293  // We only have to regenerate time-series according the settings
294  // in general data of the study.
295  logs.info() << " Only the preprocessors are enabled.";
296 
297  regenerateTimeSeries(study, pResultWriter, pDurationCollector);
298 
299  // Destroy the TS Generators if any
300  // It will export the time-series into the output at the same time
301  TSGenerator::DestroyAll(study);
302  }
303  else
304  {
305  // Export ts-numbers into output
306  TimeSeriesNumbers::StoreTimeSeriesNumbersIntoOuput(study, pResultWriter);
307 
308  if (not ImplementationType::simulationBegin())
309  {
310  return;
311  }
312  // Allocating the memory
313  ImplementationType::variables.simulationBegin();
314 
315  // For beauty
316  logs.info();
317 
318  // Launching the simulation for all years
319  logs.info() << "MC-Years : [" << (study.runtime.rangeLimits.year[Data::rangeBegin] + 1)
320  << " .. " << (1 + study.runtime.rangeLimits.year[Data::rangeEnd])
321  << "], total: " << study.runtime.rangeLimits.year[Data::rangeCount];
322 
323  // Current state
324  std::vector<Variable::State> state(pNbMaxPerformedYearsInParallel, Variable::State(study));
325  // Initializing states for parallel actually performed years
326  for (uint numSpace = 0; numSpace != pNbMaxPerformedYearsInParallel; ++numSpace)
327  {
328  ImplementationType::initializeState(state[numSpace], numSpace);
329  }
330 
331  uint finalYear = 1 + study.runtime.rangeLimits.year[Data::rangeEnd];
332  {
333  pDurationCollector("mc_years")
334  << [finalYear, &state, this] { loopThroughYears(0, finalYear, state); };
335  }
336  // Destroy the TS Generators if any
337  // It will export the time-series into the output in the same time
338  TSGenerator::DestroyAll(study);
339 
340  // Post operations
341  pDurationCollector("post_processing") << [this] { ImplementationType::simulationEnd(); };
342 
343  ImplementationType::variables.simulationEnd();
344 
345  // Spatial clusters
346  // Notifying all variables to perform the final spatial clusters.
347  // This must be done only when all variables have finished to compute their
348  // own data.
349  ImplementationType::variables.simulationEndSpatialAggregates(ImplementationType::variables);
350  }
351 }

353 template<class ImplementationType>
354 void ISimulation<ImplementationType>::writeResults(bool synthesis, uint year, uint numSpace)
355 {
356  using namespace Yuni;
357 
358  // The writer might need the job queue, after it's been stopped
359  // this is the case e.g if synthesis == true (writing mc-all)
360  // Don't restart the queue if the writer doesn't need it
361 
362  assert(!settings.noOutput);
363  assert(!settings.tsGeneratorsOnly);
364 
365  if (!pNbYearsReallyPerformed)
366  {
367  logs.info();
368  logs.info() << "The writing of the output results is disabled.";
369  logs.info();
370  }
371  else
372  {
373  if (synthesis)
374  {
375  const auto& parameters = study.parameters;
376  if (not parameters.synthesis) // disabled by parameters
377  {
378  logs.info() << "The simulation synthesis is disabled.";
379  return;
380  }
381  }
382 
383  // The target folder
384  String newPath;
385  newPath << ImplementationType::Name() << Yuni::IO::Separator;
386  if (synthesis)
387  {
388  newPath << "mc-all";
389  }
390  else
391  {
392  CString<10, false> tmp;
393  tmp = (year + 1);
394  newPath << "mc-ind" << Yuni::IO::Separator << "00000";
395  newPath.overwriteRight(tmp);
396  }
397 
398  // Dumping
399  ImplementationType::variables.exportSurveyResults(synthesis,
400  newPath,
401  numSpace,
402  pResultWriter);
403  }
404 }

417 template<class ImplementationType>
418 void ISimulation<ImplementationType>::loopThroughYears(uint firstYear,
419  uint endYear,
420  std::vector<Variable::State>& state)
421 {
422  assert(endYear <= study.parameters.nbYears);
423 
424  // Init random hydro
425  MersenneTwister randomHydroGenerator;
426  randomHydroGenerator.reset(study.parameters.seed[Data::seedHydroManagement]);
427 
428  // List of parallel years sets
429  std::vector<setOfParallelYears> setsOfParallelYears;
430 
431  // Number of threads to perform the jobs waiting in the queue
432  pQueueService->maximumThreadCount(pNbMaxPerformedYearsInParallel);
433 
434  regenerateTimeSeries(study, pResultWriter, pDurationCollector);
435  HydroInputsChecker hydroInputsChecker(study);
436  logs.info() << " Doing hydro validation";
437 
438  // Selecting the years to be performed
439  std::map<unsigned, bool> yearsFailed;
440  std::map<unsigned, bool> isYearPerformed;
441  pNbYearsReallyPerformed = 0;
442  for (uint year = firstYear; year < endYear; ++year)
443  {
444  isYearPerformed[year] = study.parameters.yearsFilter[year];
445  if (study.parameters.yearsFilter[year])
446  {
447  ++pNbYearsReallyPerformed;
448  }
449  else
450  {
451  logs.debug() << "Ignoring year " << year + 1 << ": not in the playlist";
452  yearsFailed[year] = false;
453  }
454  }
455 
456  if (pNbYearsReallyPerformed != endYear)
457  {
458  logs.info() << "Playlist detected, " << pNbYearsReallyPerformed
459  << " years to be performed out of " << endYear << " years.";
460  }
461 
462  logs.info() << " Starting the simulation";
463 
464  // Related to annual costs statistics (printed in output into separate files)
465  pAnnualStatistics.setNbPerformedYears(pNbYearsReallyPerformed);
466 
467  // Container for random numbers of parallel years (to be executed or not)
468  randomNumbers randomForParallelYears(pNbYearsReallyPerformed,
469  study.parameters.power.fluctuations);
470 
471  randomForParallelYears.allocate(study);
472  randomForParallelYears.compute(study, endYear, isYearPerformed, randomHydroGenerator);
473 
474  // hydro checks
475  for (uint year = firstYear; year < endYear; ++year)
476  {
477  if (study.parameters.yearsFilter[year])
478  {
479  hydroInputsChecker.Execute(year);
480  }
481  }
482  hydroInputsChecker.CheckForErrors();
483 
484  NumSpaceManager numspaceManager(pNbMaxPerformedYearsInParallel);
485 
486  Concurrency::FutureSet results;
487  std::mutex aggregationMutex;
488  for (uint year = firstYear; year < endYear; ++year)
489  {
490  if (study.parameters.yearsFilter[year])
491  {
492  // If the year has not to be rerun, we skip the computation of the year.
493  // Note that, when we enter for the first time in the "for" loop, all years of the set
494  // have to be rerun (meaning : they must be run once). if(!batch.yearsFailed[y])
495  // continue;
496 
497  auto task = std::make_shared<yearJob<ImplementationType>>(this,
498  year,
499  yearsFailed,
500  numspaceManager,
501  randomForParallelYears,
502  study,
503  state,
504  pYearByYear,
505  pDurationCollector,
506  pResultWriter,
507  simulationObserver_.get(),
508  aggregationMutex);
509  results.add(Concurrency::AddTask(*pQueueService, task));
510  }
511  }
512 
513  pQueueService->start();
514 
515  pQueueService->wait(Yuni::qseIdle);
516  pQueueService->stop();
517  if (!study.parameters.noOutput)
518  {
519  aggregateAndWriteSimulationTables();
520  }
521  results.join();
522  pResultWriter.flush();
523  // On regarde si au moins une année du lot n'a pas trouvé de solution
524  for (auto& [year, failed]: yearsFailed)
525  {
526  // Si une année du lot d'années n'a pas trouvé de solution, on arrête tout
527  if (failed)
528  {
529  std::ostringstream msg;
530  msg << "Year " << year + 1 << " has failed in the previous set of parallel year.";
531  throw FatalError(msg.str());
532  }
533  }
534 
535  // Set to zero the random numbers of all parallel years
536  randomForParallelYears.reset();
537 
538  // Writing annual costs statistics
539  pAnnualStatistics.endStandardDeviations();
540  pAnnualStatistics.writeToOutput(pResultWriter);
541 }

553 template<class ImplementationType>
554 void ISimulation<ImplementationType>::aggregateAndWriteSimulationTables()
555 {
556  std::lock_guard lock(buffersMutex_);
557  std::string globalFirstBuffer;
558  std::string globalSecondBuffer;
559  // dans l'ordre des années
560  for (uint year = 0; year < study.parameters.nbYears; ++year)
561  {
562  auto it = yearSimulationBuffers_.find(year);
563  if (it != yearSimulationBuffers_.end())
564  {
565  globalFirstBuffer += it->second.first;
566  globalSecondBuffer += it->second.second;
567  }
568  }
569  const auto header = ImplementationType::getSimulationTableHeader() + "\n";
570  if (!globalFirstBuffer.empty())
571  {
572  std::string writerEntry = header + std::move(globalFirstBuffer);
573  pResultWriter.addEntryFromBuffer("simulation_table--optim-nb-1.csv", writerEntry);
574  }
575  if (!globalSecondBuffer.empty())
576  {
577  std::string writerEntry = header + std::move(globalSecondBuffer);
578  pResultWriter.addEntryFromBuffer("simulation_table--optim-nb-2.csv", writerEntry);
579  }
580 
581  yearSimulationBuffers_.clear();
582 }
