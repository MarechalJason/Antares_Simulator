352  template<class V, class SetT>
353  void internalSpatialAggregate(V& allVars, uint year, const SetT& set)
354  {
355  typedef typename VCardType::VCardOrigin VCardOrigin;
356  // Reset the results
357  VariableAccessorType::Reset(pValuesForTheCurrentYear[0]);
358 
359  // Make the spatial cluster
360  if (!set.empty())
361  {
362  auto end = set.end();
363  for (auto i = set.begin(); i != end; ++i)
364  {
365  allVars.template computeSpatialAggregateWith<
366  typename VCardType::VCardForSpatialAggregate> //<typename VCardType::VCardOrigin>
367  (pValuesForTheCurrentYear[0], *i /* the current area */, 0);
368  }
369 
370  // The spatial cluster may be an average
371  if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateAverage)
372  {
373  VariableAccessorType::MultiplyHourlyResultsBy(pValuesForTheCurrentYear[0],
374  1. / set.size());
375  }
376  // The spatial cluster may be an average
377  if (VCardType::VCardOrigin::spatialAggregate
378  & Category::spatialAggregateSumThen1IfPositive)
379  {
380  VariableAccessorType::SetTo1IfPositive(pValuesForTheCurrentYear[0]);
381  }
382  if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateOr)
383  {
384  VariableAccessorType::Or(pValuesForTheCurrentYear[0]);
385  }
386  }
387  else
388  {
389  assert(!set.empty() && "The set should not be empty at this point");
390  }
391 
392  // Compute all statistics for the current year (daily,weekly,monthly,...)
393  VariableAccessorType::template ComputeStatistics<VCardOrigin>(pValuesForTheCurrentYear[0]);
394  VariableAccessorType::ComputeSummary(pValuesForTheCurrentYear[0],
395  AncestorType::pResults,
396  year);
397  }

399  template<class V, class SetT>
400  void internalSpatialAggregateForCurrentYear(V& allVars, const SetT& set, uint numSpace)
401  {
402  typedef typename VCardType::VCardOrigin VCardOrigin;
403  // Reset the results
404  VariableAccessorType::Reset(pValuesForTheCurrentYear[numSpace]);
405 
406  // Make the spatial cluster
407  if (!set.empty())
408  {
409  // We compute the district's (or set) hourly values by looping over district's areas
410  // and calling computeSpatialAggregateWith on each area.
411  // The aggregated results is stored in pValuesForTheCurrentYear[numSpace].
412  auto end = set.end();
413  for (auto i = set.begin(); i != end; ++i)
414  {
415  // computeSpatialAggregateWith is called with VCardType::VCardForSpatialAggregate
416  // as a template parameter.
417  // Note that VCardType represents the vcard of the current spatial aggregate's
418  // variable (template parameter VarT).
419  // For most variables VCardType::VCardOrigin and
420  // VCardType::VCardForSpatialAggregate are identical vcard types.
421  // For example : for var OverallCost, VCardOverallCost::VCardForSpatialAggregate
422  // is simply VCardOverallCost itself.
423  // But var LOLD is different : VCardLOLD::VCardForSpatialAggregate is actually
424  // VCardUnsupliedEnergy.
425  // This means that, here :
426  // - for OverallCost, we sum the hourly values of var OverallCost for each
427  // district's areas.
428  allVars.template computeSpatialAggregateWith<
429  typename VCardType::VCardForSpatialAggregate>(pValuesForTheCurrentYear[numSpace],
430  *i /* the current area */,
431  numSpace);
432  }
433 
434  // The spatial cluster may be an average
435  if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateAverage)
436  {
437  VariableAccessorType::MultiplyHourlyResultsBy(pValuesForTheCurrentYear[numSpace],
438  1. / set.size());
439  }
440  // The spatial cluster may be an average
441  if (VCardType::VCardOrigin::spatialAggregate
442  & Category::spatialAggregateSumThen1IfPositive)
443  {
444  VariableAccessorType::SetTo1IfPositive(pValuesForTheCurrentYear[numSpace]);
445  }
446  if (VCardType::VCardOrigin::spatialAggregate & Category::spatialAggregateOr)
447  {
448  VariableAccessorType::Or(pValuesForTheCurrentYear[numSpace]);
449  }
450  }
451  else
452  {
453  assert(!set.empty() && "The set should not be empty at this point");
454  }
455 
456  // Compute all statistics for the current year (daily,weekly,monthly,...)
457  VariableAccessorType::template ComputeStatistics<VCardOrigin>(
458  pValuesForTheCurrentYear[numSpace]);
459  }
