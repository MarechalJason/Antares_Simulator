19 // Represents an output variable (wears the same name) and mainly answers the question :
20 // Is the real variable printed in all output reports ? Or is it not printed in any report ?
21 class VariablePrintInfo final
22 {
23 public:
24  VariablePrintInfo(uint dataLvl, uint fileLvl);
25  ~VariablePrintInfo() = default;
26 
27  // Do we enable or disable variable's print in output reports ?
28  void enablePrint(bool b);
29  bool isPrinted() const;
30  void reverse();
31 
32  uint getMaxColumnsCount();
33  void setMaxColumns(uint maxColumnsNumber);
34 
35  bool isPrintedOnDataLevel(uint dataLevel) const
36  {
37  return dataLevel_ & dataLevel;
38  }
39 
40  bool isPrintedOnFileLevel(uint fileLevel) const
41  {
42  return fileLevel_ & fileLevel;
43  }
44 
45 private:
46  // Is the variable printed ?
47  bool to_be_printed_ = true;
48 
49  // The number of columns the output variable takes in a SYNTHESIS report.
50  // Recall that synthesis reports always contain more columns than
51  // any other reports (for instance year-by-year reports)
52  uint maxNumberColumns_ = 0;
53 
54  // Which reports the output variable has columns in ?
55  // Example : areas/values-<time-interval>.txt
56  // dataLevel can be : areas, links, bindingConstraint
57  // fileLevel can be : values-<time-interval>.txt, details-<time-interval>.txt,
58  // id-<time-interval>.txt, ...
59  uint dataLevel_ = 0;
60  uint fileLevel_ = 0;
61 };
